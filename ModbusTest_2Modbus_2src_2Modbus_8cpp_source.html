371int Modbus::CheckCRC(byte *buf, int len)
372{
373 int nominal = 0xA001;
374 int crc = 0xFFFF;
375 unsigned char pos,i;
376
377 for ( pos = 0; pos < len; pos++) {
378 crc ^= (unsigned int)buf[pos]; // XOR byte into least sig. byte of crc
379
380 for (i = 8; i != 0; i--) { // Loop over each bit
381 if ((crc & 0x0001) != 0) { // If the LSB is set
382 crc >>= 1; // Shift right and XOR 0xA001
383 crc ^= nominal;
384 }
385 else // Else LSB is not set
386 crc >>= 1; // Just shift right
387 }
388 }
389 // Note, this number has low and high bytes swapped, so use it accordingly (or swap bytes)
390 return crc;
391}
