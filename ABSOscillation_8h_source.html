28 void forceOffset(DAP_calculationVariables_st* calcVars_st, uint8_t absPattern, uint8_t absForceOrTarvelBit, float * absForceOffset, float * absPosOffset) {
29
30
31 long timeNowMillis = millis();
32 float timeSinceTrigger = (timeNowMillis - _timeLastTriggerMillis);
33 float absForceOffset_local = 0;
34
35 if (timeSinceTrigger > ABS_ACTIVE_TIME_PER_TRIGGER_MILLIS)
36 {
37 _absTimeMillis = 0;
38 absForceOffset = 0;
39 }
40 else
41 {
42 _absTimeMillis += timeNowMillis - _lastCallTimeMillis;
43 float absTimeSeconds = _absTimeMillis / 1000.0f;
44
45 // abs amplitude
46 float absAmp_fl32 = 0;
47 switch (absForceOrTarvelBit) {
48 case 0:
49 absAmp_fl32 = calcVars_st->absAmplitude;
50 break;
51 case 1:
52 absAmp_fl32 = calcVars_st->stepperPosRange * calcVars_st->absAmplitude / 100.;
53 break;
54 default:
55 break;
56 }
57
58
59
60 switch (absPattern) {
61 case 0:
62 // sine wave pattern
63 absForceOffset_local = absAmp_fl32 * sin(2 * PI * calcVars_st->absFrequency * absTimeSeconds);
64 break;
65 case 1:
66 // sawtooth pattern
67 if (calcVars_st->absFrequency > 0)
68 {
69 absForceOffset_local = absAmp_fl32 * fmod(absTimeSeconds, 1.0 / (float)calcVars_st->absFrequency) * (float)calcVars_st->absFrequency;
70 }
71 break;
72 default:
73 break;
74 }
75
76 switch (absForceOrTarvelBit) {
77 case 0:
78 *absForceOffset = absForceOffset_local;
79 *absPosOffset = 0;
80 break;
81 case 1:
82 *absForceOffset = 0;
83 *absPosOffset = absForceOffset_local;
84 break;
85 default:
86 break;
87 }
88
89 }
90
91 _lastCallTimeMillis = timeNowMillis;
92
93 return;
94
95 }

235
236MovingAverageFilter::MovingAverageFilter(unsigned int newDataPointsCount)
237{
238 k = 0; //initialize so that we start to write at index 0
239 if (newDataPointsCount < MAX_DATA_POINTS)
240 dataPointsCount = newDataPointsCount;
241 else
242 dataPointsCount = MAX_DATA_POINTS;
243
244 for (i = 0; i < dataPointsCount; i++)
245 {
246 values[i] = 0; // fill the array with 0's
247 }
248}
249
250float MovingAverageFilter::process(float in)
251{
252 out = 0;
253
254 values[k] = in;
255 k = (k + 1) % dataPointsCount;
256
257 for (i = 0; i < dataPointsCount; i++)
258 {
259 out += values[i];
260 }
261
262 float retValue= 0;
263 if (dataPointsCount> 0)
264 {
265 retValue= out / dataPointsCount;
266 }
267
268 return retValue;
269}
