204 // see https://github.com/ChrGri/DIY-Sim-Racing-FFB-Pedal/wiki/Movement-control-strategies#mpc
205
206
207 /*
208 This closed-loop control strategy models the foot as a spring with a certain stiffness k1.
209 The force resulting from that model is F1 = k1 * x.
210 To find the servo target position:
211 1) A line with the slope -k1 at the point of the loadcell reading & current position is drawn.
212 2) The intersection with the force-travel curve gives the target position
213
214 Since the force-travel curve might be non-linear, Newtons method is used to numerically find the intersection point.
215 f(x_n) = -k1 * x + b - forceCurve(x) = 0
216 x_n+1 = x_n - f(x_n) / f'(x_n)
217 whereas x_n is the servo position at iteration n
218 f(x_n) = -k1 * x + b - forceCurve(x)
219 f'(x_n) = -k1 - d(forceCurve(x)) / dx
220 */
221
222 // get current stepper position
223 float stepperPos = stepper->getCurrentPosition();
224
225 // add velocity feedforward
226 stepperPos += changeVelocity * config_st->payLoadPedalConfig_.PID_velocity_feedforward_gain;
227
228 // motion corrected loadcell reading
229 float loadCellReadingKg_corrected = loadCellReadingKg;
230 //loadCellReadingKg_corrected += config_st->payLoadPedalConfig_.MPC_1st_order_gain * stepper_vel_filtered_fl32 / STEPS_PER_MOTOR_REVOLUTION / 10;// + stepper_accel_filtered_fl32;
231 //loadCellReadingKg_corrected += config_st->payLoadPedalConfig_.MPC_1st_order_gain * stepper_accel_filtered_fl32 / STEPS_PER_MOTOR_REVOLUTION / 10;// + stepper_accel_filtered_fl32;
232
233
234 // set initial guess
235 float stepperPos_initial = stepperPos;
236
237 // Find the intersections of the force curve and the foot model via Newtons-method
238 #define MAX_NUMBER_OF_NEWTON_STEPS 5
239 for (uint8_t iterationIdx = 0; iterationIdx < MAX_NUMBER_OF_NEWTON_STEPS; iterationIdx++)
240 {
241 //float stepperPosFraction = stepper->getCurrentPositionFraction();
242 float stepperPosFraction = stepper->getCurrentPositionFractionFromExternalPos( stepperPos );
243
244 // clamp the stepper position to prevent problems with the spline
245 float x_0 = constrain(stepperPosFraction, 0, 1);
246
247 // get force and force gradient of force vs travel curve
248 float loadCellTargetKg = forceCurve->EvalForceCubicSpline(config_st, calc_st, x_0);
249 float gradient_force_curve_fl32 = forceCurve->EvalForceGradientCubicSpline(config_st, calc_st, x_0, false);
250
251 // apply effect force offset
252 loadCellTargetKg -=absForceOffset_fl32;
253
254 // how many mm movement to order if 1kg of error force is detected
255 // this can be tuned for responsiveness vs oscillation
256 float mm_per_motor_rev = config_st->payLoadPedalConfig_.spindlePitch_mmPerRev_u8;//TRAVEL_PER_ROTATION_IN_MM;
257 float steps_per_motor_rev = STEPS_PER_MOTOR_REVOLUTION;
258
259 // foot spring stiffness
260 float d_f_d_phi = -config_st->payLoadPedalConfig_.MPC_0th_order_gain;
261
262 float move_mm_per_kg = config_st->payLoadPedalConfig_.MPC_0th_order_gain;
263 float MOVE_STEPS_FOR_1KG = (move_mm_per_kg / mm_per_motor_rev) * steps_per_motor_rev;
264
265
266
267 // make stiffness dependent on force curve gradient
268 // less steps per kg --> steeper line
269 float gradient_normalized_force_curve_fl32 = forceCurve->EvalForceGradientCubicSpline(config_st, calc_st, x_0, true);
270 gradient_normalized_force_curve_fl32 = constrain(gradient_normalized_force_curve_fl32, 0.05, 1);
271
272 float mmPerStep = 0;
273 if (steps_per_motor_rev > 0)
274 {
275 mmPerStep = mm_per_motor_rev / steps_per_motor_rev ;
276 }
277
278
279
280 // The foot is modeled to be of proportional resistance with respect to deflection. Since the deflection depends on the pedal kinematics, the kinematic must be respected here
281 // This is accomplished with the forceGain variable
282 /*float forceGain_abs = fabs( d_phi_d_x );
283 if (forceGain_abs > 0)
284 {
285 MOVE_STEPS_FOR_1KG *= fabs( forceGain );
286 }*/
287
288
289
290 // angular foot model
291 // m1 = d_f_d_x dForce / dx
292 //float m1 = d_f_d_phi * (-d_phi_d_x);
293
294
295 // Translational foot model
296 // given in kg/step
297 float m1 = d_f_d_phi * (-d_x_hor_d_phi) * (-d_phi_d_x) * mmPerStep;
298
299 // smoothen gradient with force curve gradient since it had better results w/ clutch pedal characteristic
300 //m1 /= gradient_normalized_force_curve_fl32;
301
302 // gradient of the force curve
303 // given in kg/step
304 float m2 = gradient_force_curve_fl32;
305
306 // Newton update
307 float denom = m1 - m2;
308 if ( fabs(denom) > 0 )
309 {
310 float stepUpdate = ( loadCellReadingKg_corrected - loadCellTargetKg) / ( denom );
311
312 // smoothen update with force curve gradient since it had better results w/ clutch pedal characteristic
313 stepUpdate *= gradient_normalized_force_curve_fl32;
314
315 stepperPos -= stepUpdate;
316 }
317 }
318
319 // limit the position update
320 float deltaMax = 0.5 * (float)(calc_st->stepperPosMax - calc_st->stepperPosMin);
321 int32_t posStepperNew = constrain(stepperPos, stepperPos_initial-deltaMax, stepperPos_initial+deltaMax );
322
323 // clamp target position to range
324 posStepperNew = constrain(posStepperNew,calc_st->stepperPosMin,calc_st->stepperPosMax );
325
326 return posStepperNew;
327}
