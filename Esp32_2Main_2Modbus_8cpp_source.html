415int Modbus::CheckCRC(byte *buf, int len)
416{
417 int nominal = 0xA001;
418 int crc = 0xFFFF;
419 unsigned char pos,i;
420
421 for ( pos = 0; pos < len; pos++) {
422 crc ^= (unsigned int)buf[pos]; // XOR byte into least sig. byte of crc
423
424 for (i = 8; i != 0; i--) { // Loop over each bit
425 if ((crc & 0x0001) != 0) { // If the LSB is set
426 crc >>= 1; // Shift right and XOR 0xA001
427 crc ^= nominal;
428 }
429 else // Else LSB is not set
430 crc >>= 1; // Just shift right
431 }
432 }
433 // Note, this number has low and high bytes swapped, so use it accordingly (or swap bytes)
434 return crc;
435}
